Thrift

安装

brew install thrift

声明数据格式

sample.thrift:这里我们定义一个服务Service在其中声明了一个echo方法

 
 
service Service {
  string echo(1:string msg)
}

编译

thrift -r --gen js:node sample.thrift

会在根目录下生成gen-nodejs文件夹,里面有对应的数据类型和服务声明

-r标示会编译当前文件include进来的其他thrift文件

服务端

需要实现echo:

 
xxxxxxxxxx
 
var thrift = require("thrift");
var Service = require("./gen-nodejs/Service");
var sample_types = require("./gen-nodejs/sample_types");
var server = thrift.createServer(Service, {
    echo: function (msg) {
        return msg;
    }
});
server.listen(9090);

客户端

调用echo:

 
​x
 
var thrift = require('thrift');
var Service = require("./gen-nodejs/Service");
var sample_types = require("./gen-nodejs/sample_types");
​
var connection = thrift.createConnection("localhost", 9090);
var client = thrift.createClient(Service, connection);
​
client.echo("hello thrift").then(function(res){
    console.log(res);
});
​
connection.on('error', function(err) {
    console.error(err);
});

结构体

thrift
 
xxxxxxxxxx
 
struct UserInfo {
    1: i32 uid
    2: string name
}
service Service {
  string echo(1:string msg)
  UserInfo getUserInfo(1:string token)
}

可以用关键字 required或者optional来声明字段是否可选 

 
xxxxxxxxxx
 
struct UserInfo {
    1: required i32 uid
    2: optional string name
}
服务端
 
xxxxxxxxxx
 
getUserInfo: function(token) {
  var userInfo = new sample_types.UserInfo();
  userInfo.uid = 123131212, userInfo.name = "sunshengda";
  return userInfo;
}
客户端
 
xxxxxxxxxx
 
client.getUserInfo("asdjkladkasdja").then(function(res){
    console.log(res);
});

异常

thrift
 
xxxxxxxxxx
 
exception InvalidOperation {
  1: i32 whatOp,
  2: string why
}
service Service {
  Container geContainer() throws (1:InvalidOperation ouch)
}

xxxxxxxxxx
 
var exceptions = new sample_types.InvalidOperation();

throw exceptions;
